=== CONCEPT: incremental_input_ingest_from_archives ===
Description: Idempotently extract new .tar.gz inputs into inputs/ and append only previously-untracked .zip jobs into file_list.txt.
Category: adapters
Invariants:
  - "Never re-add a .zip already tracked: compute set-difference between on-disk zips and file_list.txt using sorted absolute paths + comm."
  - "Preserve nested directory structure under inputs/ when extracting tarballs."
  - "file_list.txt format must remain: 'zip_abspath, zip_relpath_no_ext' (comma+space delimiter)."
Interface_inputs: {{INPUT_TARBALLS_OR_ARGS}}, {{INPUTS_DIR}}, {{FILE_LIST}}
Interface_outputs: Appended file_list.txt entries for new zips; temp files removed; NEW_COUNT printed
Scheduler: any
Tool: any
Infrastructure: shared_fs
Tested_by: golddigr
Notes: From add_data.sh steps 1–3; safe to run multiple times.

```bash
#!/bin/bash
set -e

SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
cd "$SCRIPT_DIR"

INPUTS_DIR="{{INPUTS_DIR:-inputs}}"
FILE_LIST="{{FILE_LIST:-file_list.txt}}"

# Determine tarballs to process
if [ $# -gt 0 ]; then
  INPUT_TARBALLS="$@"
else
  INPUT_TARBALLS=$(ls *.tar.gz 2>/dev/null | grep -v 'results_package' | grep -v 'backup_' || true)
fi

if [ -z "$INPUT_TARBALLS" ]; then
  echo "Error: No *.tar.gz files found."
  exit 1
fi

# Preflight
if [ ! -f "$FILE_LIST" ]; then
  echo "Error: $FILE_LIST not found. Initialize workflow first."
  exit 1
fi
if [ ! -d "$INPUTS_DIR" ] || [ ! -d "results" ]; then
  echo "Error: inputs/ or results/ directory missing. Initialize workflow first."
  exit 1
fi

# Step 1: Extract archives into inputs/
for TARBALL in $INPUT_TARBALLS; do
  if [ ! -f "$TARBALL" ]; then
    echo "Warning: $TARBALL not found, skipping"
    continue
  fi
  tar -xzf "$TARBALL" -C "$INPUTS_DIR"/
done

# Step 2: Find NEW zip files not already tracked
INPUTS_ABS="$(cd "$INPUTS_DIR" && pwd)"

ALL_ZIPS_FILE=".all_zips_$$.txt"
find "$INPUTS_ABS" -name "*.zip" -type f | sort > "$ALL_ZIPS_FILE"

TRACKED_ZIPS_FILE=".tracked_zips_$$.txt"
awk -F', ' '{print $1}' "$FILE_LIST" | sort > "$TRACKED_ZIPS_FILE"

NEW_ZIPS_FILE=".new_zips_$$.txt"
comm -23 "$ALL_ZIPS_FILE" "$TRACKED_ZIPS_FILE" > "$NEW_ZIPS_FILE"

rm -f "$ALL_ZIPS_FILE" "$TRACKED_ZIPS_FILE"

NEW_COUNT=$(wc -l < "$NEW_ZIPS_FILE")
if [ "$NEW_COUNT" -eq 0 ]; then
  echo "No new .zip files found; all zips already tracked."
  rm -f "$NEW_ZIPS_FILE"
  exit 0
fi

# Convert new zip paths to file_list format: "abspath, relpath_no_ext"
NEW_LIST_FILE=".new_entries_$$.txt"
while IFS= read -r zipfile; do
  relpath="${zipfile#$INPUTS_ABS/}"
  relpath_no_ext="${relpath%.zip}"
  echo "$zipfile, $relpath_no_ext"
done < "$NEW_ZIPS_FILE" > "$NEW_LIST_FILE"
rm -f "$NEW_ZIPS_FILE"

# Step 3: Append and mirror directories
cat "$NEW_LIST_FILE" >> "$FILE_LIST"
(cd "$INPUTS_DIR" && find . -type d -exec mkdir -p "../results/{}" \;)

rm -f "$NEW_LIST_FILE"

echo "Appended $NEW_COUNT new entries to $FILE_LIST"
```

=== CONCEPT: bulk_config_patch_with_sed ===
Description: Idempotent in-place edits across workflow files to change ORCA maxcore, HTCondor request_memory, and health escalation ladder.
Category: monitoring
Invariants:
  - "Edits must be safe to re-run (detect already-updated values)."
  - "RESOURCE_LEVELS block replacement must match the script’s existing block boundaries."
Interface_inputs: {{RUN_ORCA_SCRIPT}}, {{SUBMIT_FILE}}, {{HEALTH_MONITOR}}
Interface_outputs: Updated files on disk; nonzero ERRORS count if unexpected patterns
Scheduler: htcondor
Tool: orca
Infrastructure: shared_fs
Tested_by: golddigr
Notes: From apply_memory_fix.sh steps 1–3 (as-is; includes warnings when patterns don’t match).

```bash
#!/bin/bash
set -e

RUN_ORCA_SCRIPT="{{RUN_ORCA_SCRIPT:-run_orca_wbo.sh}}"
SUBMIT_FILE="{{SUBMIT_FILE:-batch_job.submit}}"
HEALTH_MONITOR="{{HEALTH_MONITOR:-monitor_health.sh}}"

ERRORS=0

# Step 1: %maxcore 2500 -> 1000
if [ ! -f "$RUN_ORCA_SCRIPT" ]; then
  echo "Error: $RUN_ORCA_SCRIPT not found"
  ERRORS=$((ERRORS + 1))
else
  if grep -q '%maxcore 2500' "$RUN_ORCA_SCRIPT"; then
    sed -i 's/%maxcore 2500/%maxcore 1000/g' "$RUN_ORCA_SCRIPT"
  elif grep -q '%maxcore 1000' "$RUN_ORCA_SCRIPT"; then
    :
  else
    echo "Warning: Could not find %maxcore line"
    ERRORS=$((ERRORS + 1))
  fi
fi

# Step 2: request_memory 12 GB -> 6 GB
if [ ! -f "$SUBMIT_FILE" ]; then
  echo "Error: $SUBMIT_FILE not found"
  ERRORS=$((ERRORS + 1))
else
  if grep -q 'request_memory = 12 GB' "$SUBMIT_FILE"; then
    sed -i 's/request_memory = 12 GB/request_memory = 6 GB/g' "$SUBMIT_FILE"
  elif grep -q 'request_memory = 6 GB' "$SUBMIT_FILE"; then
    :
  else
    echo "Warning: Unexpected request_memory value"
    grep 'request_memory' "$SUBMIT_FILE" 2>/dev/null || true
    ERRORS=$((ERRORS + 1))
  fi
fi

# Step 3: Replace RESOURCE_LEVELS block
if [ ! -f "$HEALTH_MONITOR" ]; then
  echo "Error: $HEALTH_MONITOR not found"
  ERRORS=$((ERRORS + 1))
else
  if grep -q '"12288:4"' "$HEALTH_MONITOR" || grep -q '"16384:4"' "$HEALTH_MONITOR"; then
    sed -i '/^RESOURCE_LEVELS=(/,/^)/c\
RESOURCE_LEVELS=(\
    "6144:4"     # Level 0: 6 GB, 4 CPUs  (default)\
    "12288:6"    # Level 1: 12 GB, 6 CPUs\
    "20480:8"    # Level 2: 20 GB, 8 CPUs\
    "32768:10"   # Level 3: 32 GB, 10 CPUs (MAX)\
)' "$HEALTH_MONITOR"
  elif grep -q '"6144:4"' "$HEALTH_MONITOR"; then
    :
  else
    echo "Warning: Could not find resource levels block"
    ERRORS=$((ERRORS + 1))
  fi

  sed -i 's/Level 0: 12 GB/Level 0: 6 GB/g; s/Level 0: 16 GB/Level 0: 6 GB/g' "$HEALTH_MONITOR"
  sed -i 's/Level 1: 20 GB/Level 1: 12 GB/g; s/Level 1: 24 GB/Level 1: 12 GB/g' "$HEALTH_MONITOR"
  sed -i 's/Level 2: 32 GB/Level 2: 20 GB/g' "$HEALTH_MONITOR"
  sed -i 's/Level 3: 40 GB/Level 3: 32 GB/g' "$HEALTH_MONITOR"
  sed -i 's/L0: [0-9]*GB:[0-9]* .* L1: [0-9]*GB:[0-9]* .* L2: [0-9]*GB:[0-9]* .* L3: [0-9]*GB:[0-9]* (MAX)/L0: 6GB:4  →  L1: 12GB:6  →  L2: 20GB:8  →  L3: 32GB:10 (MAX)/g' "$HEALTH_MONITOR"
fi

echo "ERRORS=$ERRORS"
exit 0
```

  - "Never delete originals unless tar -tzf verification succeeds."
Interface_inputs: {{RESULTS_DIR}}, {{DEST_DIR}}, {{MANIFEST_DIR}}, {{FILE_LIST_ABS_NEWLINE}}
Interface_outputs: backup_*.tar.gz in DEST_DIR; manifest_*.txt in MANIFEST_DIR; originals removed

Notes: Extracted from backup_results.sh create_backup().

```bash
#!/bin/bash
set -e

RESULTS_DIR="{{RESULTS_DIR}}"
DEST_DIR="{{DEST_DIR}}"
MANIFEST_DIR="{{MANIFEST_DIR:-.backup_manifests}}"
FILE_LIST_ABS_NEWLINE="{{FILE_LIST_ABS_NEWLINE}}"

mkdir -p "$DEST_DIR" "$MANIFEST_DIR"

file_count=$(echo "$FILE_LIST_ABS_NEWLINE" | grep -c . || true)
timestamp=$(date '+%Y%m%d_%H%M%S')
backup_name="backup_${timestamp}_n${file_count}.tar.gz"
backup_tmp="$DEST_DIR/${backup_name}.tmp"
backup_final="$DEST_DIR/${backup_name}"
manifest_file="$MANIFEST_DIR/manifest_${timestamp}.txt"

echo "# Backup manifest: $backup_name" > "$manifest_file"
echo "# Created: $(date)" >> "$manifest_file"
echo "# Status: IN_PROGRESS" >> "$manifest_file"
echo "# Destination: $backup_final" >> "$manifest_file"
echo "# Files:" >> "$manifest_file"
echo "$FILE_LIST_ABS_NEWLINE" >> "$manifest_file"

# Create tar.gz preserving structure relative to RESULTS_DIR
if ! echo "$FILE_LIST_ABS_NEWLINE" | tar -czf "$backup_tmp" -C "$RESULTS_DIR" --files-from=<(
  echo "$FILE_LIST_ABS_NEWLINE" | while IFS= read -r f; do
    echo "${f#$RESULTS_DIR/}"
  done
) 2>/dev/null; then
  echo "# Status: FAILED_CREATE" >> "$manifest_file"
  rm -f "$backup_tmp"
  exit 1
fi

# Verify
if ! tar -tzf "$backup_tmp" >/dev/null 2>&1; then
  echo "# Status: FAILED_VERIFY" >> "$manifest_file"
  rm -f "$backup_tmp"
  exit 1
fi

# Atomic rename
mv "$backup_tmp" "$backup_final"

# Delete originals
deleted=0
failed=0
while IFS= read -r file; do
  [ -z "$file" ] && continue
  if rm -f "$file" 2>/dev/null; then
    deleted=$((deleted + 1))
  else
    failed=$((failed + 1))
  fi
done <<< "$FILE_LIST_ABS_NEWLINE"

find "$RESULTS_DIR" -type d -empty -delete 2>/dev/null || true

sed -i 's/IN_PROGRESS/COMPLETE/' "$manifest_file"
echo "# Completed: $(date)" >> "$manifest_file"
echo "# Deleted: $deleted, Failed: $failed" >> "$manifest_file"

echo "Backup complete: $backup_final"
```

```bash
MANIFEST_DIR="{{MANIFEST_DIR:-.backup_manifests}}"

      rm -f "$manifest"
```

=== CONCEPT: capacity_aware_staged_submission_monitor ===
Description: Drip-feed chunk files to HTCondor when queue is below a max job threshold; move submitted chunks to submitted_batches/ for idempotency.
Category: scheduler
Invariants:
  - "Never resubmit a chunk already present in submitted_batches/."
  - "Gate submission on total job count < MAX_JOBS."
  - "Submit chunks in numeric order (job_1, job_2, ...)."
Interface_inputs: {{MAX_JOBS}}, {{CHECK_INTERVAL}}, {{SUBMIT_FILE}}, {{FILE_PATTERN}}
Interface_outputs: Submitted jobs; chunk files moved to submitted_batches/
Scheduler: htcondor
Tool: any
Infrastructure: shared_fs
Tested_by: golddigr
Notes: monitor_submit.sh as-is.

```bash
#!/bin/bash
set -e

MAX_JOBS={{MAX_JOBS:-100000}}
USER_ID="${USER}"
CHECK_INTERVAL={{CHECK_INTERVAL:-300}}
SUBMIT_FILE="{{SUBMIT_FILE:-batch_job.submit}}"
FILE_PATTERN="{{FILE_PATTERN:-job_*.txt}}"

mkdir -p submitted_batches

get_job_count() {
  local count
  count=$(condor_q "$USER_ID" 2>/dev/null | grep "Total for query" | awk '{print $4}')
  if [ -z "$count" ]; then
    count=$(condor_q "$USER_ID" 2>/dev/null | awk -v user="$USER_ID" '$0 ~ user {sum += $10} END {print sum+0}')
  fi
  echo "${count:-0}"
}

[ -f "$SUBMIT_FILE" ] || { echo "Error: Submit file not found: $SUBMIT_FILE"; exit 1; }

PENDING_FILES=$(ls $FILE_PATTERN 2>/dev/null | wc -l)
[ "$PENDING_FILES" -gt 0 ] || { echo "Error: No job files found matching $FILE_PATTERN"; exit 1; }

for batch_file in $(ls $FILE_PATTERN 2>/dev/null | sort -t_ -k2 -n); do
  [ -f "$batch_file" ] || continue

  if [ -f "submitted_batches/$batch_file" ]; then
    echo "Skipping $batch_file (already submitted)"
    continue
  fi

  while true; do
    current_jobs=$(get_job_count)
    echo "[$(date '+%H:%M:%S')] Queue: $current_jobs / $MAX_JOBS jobs"
    if [ "$current_jobs" -lt "$MAX_JOBS" ]; then
```

```bash
WALL_SECONDS=${WALL_SECONDS:-70200}
WATCHDOG_BUFFER={{WATCHDOG_BUFFER:-1800}}

WATCHDOG_PID=""

start_watchdog() {

    sleep "$WATCHDOG_DELAY"
    pkill -f "/opt/orca/orca" 2>/dev/null || true

    kill "$WATCHDOG_PID" 2>/dev/null || true
    wait "$WATCHDOG_PID" 2>/dev/null || true
    WATCHDOG_PID=""
```
